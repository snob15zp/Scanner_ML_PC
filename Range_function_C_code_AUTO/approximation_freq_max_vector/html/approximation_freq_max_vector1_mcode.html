1function [ f ] = approximation_freq_max_vector( Tm, Fd, Signal, f )
 2%UNTITLED13 Summary of this function goes here

 4
 5hf=0.01;
 6ff=0; % step and accuracy are optimized for speed, deviation ff reset
 7
 8while abs(hf)>1E-6              % if the step has not reached the optimal accuracy, then
 9	f=f+hf;                     % frequency is incremented by a step
10    sc=0;ss=0;                  % initial assignments to amounts
11    for i=0:Tm*Fd               % number of time array indexes Tm*Fd+1

14    end
15    s=sc^2+ss^2;                % variance function - sum of squares of vector sums
16    if s<ff                     % if the maximum value is skipped, then
17        hf=-hf/2;               % dividing by 2 and reversing the step in the opposite direction
18    end
19    ff=s;                       % the past sum is equal to the current one
20end                             % end of approximation in frequency
21
22end
